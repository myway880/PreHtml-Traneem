1 - سلمتُ نفسي في يدَيكْ
يا سيدي يسوعْ
قُدْني فتُكلاني عليكْ
كي أخدُمَ الجموعْ
كُنْ حارسي من الضَّلالْ
ومرشدي في كلِّ حالْ
والقلبَ قدِّسْ والفِعالْ
حتى أرى يسوعْ

ق. قد قلتَ قبلَ الآنْ
يا منبعَ الإحسانْ
يا متعبَ القلبِ تعالْ
أُمكُثْ معي حتى تنالْ
هيا ففي قُربي مَنالْ
للخيرِ في يسوعْ

2 - إن نغَّصتْ عيشي الهُمومْ
أوْ إنْ دنا الظلامْ
ستنجلي كلُّ الغيومْ
من مانحِ السلامْ
الربُّ حصني والأملْ
الربُّ إن قالَ فَعَلْ
والحبُّ فيه مُكتمِلْ
لذا أرى يسوعْ

ق. قد قلتَ قبلَ الآنْ

3 - ربي امتلكني واهدني
للخير يا منّان
أدعوك فارحم واروني
من نبعك الملآن
كيما أعود من جديد
أسعى وأطلب المزيد
من نعمة العهد الجديد
حتى أرى يسوع
