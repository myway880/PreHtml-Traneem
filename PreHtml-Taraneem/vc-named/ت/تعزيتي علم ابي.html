1 - تَعزيَتي عِلْمُ أبي
بحَيْرَتي في سُبُلي
فهْوَ الذي يقودُني
إلى الطريقِ الأكملِ

ق. كلُّ الصِّعابْ - في علمِ الآبْ
كلُّ مُصابْ - في عِلمِ الآبْ
ودمعُك الذي ينسابْ
أيضاً هوَ في علمِ الآبْ

2 - تعزيَتي عِلمُ أبي
بكلِّ شوكٍ في الورودْ
فهْو يُزيلُهُو لي
مِن طيبِ وردِه يجودْ

ق. كلُّ الصِّعابْ - في علمِ الآبْ
كلُّ مُصابْ - في عِلمِ الآبْ
ودمعُك الذي ينسابْ
أيضاً هوَ في علمِ الآبْ

3 - تعزيَني علمُ أبي
بكلِّ قوةِ العِدى
فهْو الذي يحفظُني
مُنتصراً إلى المدى

ق. كلُّ الصِّعابْ - في علمِ الآبْ
كلُّ مُصابْ - في عِلمِ الآبْ
ودمعُك الذي ينسابْ
