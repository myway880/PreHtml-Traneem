1 - قد غَرِقتُ في الشرورْ
في حَمأةِ الأثامْ
غابَ عن قلبي السرورْ
والأمنُ والسلامْ
فدعوت المفتدي
ربَّ الورى المنانْ
إنتشِلني مُنجِدي
يا ذا الحنانْ

ق. حُبُّ يسوعْ - حبُّ يسوعْ
من حمأةِ الذنوبْ نفسي انتشلْ
حُبُّ يسوعْ - حبُّ يسوعْ
من حمأةِ الذنوبْ نفسي انتشلْ

2 - عشت بائساً أثيمْ
وحاقَ بي الدُّجَى
فأتى الفادي الرحيمْ
بِالنورِ والرَّجا
إن قلبي مُلكُهُ
ما دمتُ في الوجودْ
فأُذيعُ حمدَهُ
ذاك الودودْ

ق. حُبُّ يسوعْ - حبُّ يسوعْ
من حمأةِ الذنوبْ نفسي انتشلْ
حُبُّ يسوعْ - حبُّ يسوعْ
من حمأةِ الذنوبْ نفسي انتشلْ

3 - يا غريقاً في الشرورْ
إستَنجِدِ الإلهْ
إنه الربُّ الغفورْ
يدعوكَ، للنجاهْ
منقذاً غرقَى الأثامْ
من لُجَّةِ المياهْ
وحدُه فادي الأنامْ
رَجا الخطاهْ

ق. حُبُّ يسوعْ - حبُّ يسوعْ
من حمأةِ الذنوبْ نفسي انتشلْ
حُبُّ يسوعْ - حبُّ يسوعْ
من حمأةِ الذنوبْ نفسي انتشلْ
