1 - كنتُ مديونَ العَلي
خالقِ الكلِّ
فقضى ديني الذي
ماتَ من أجلي

ق. قد قضى ديني
كلَّه الحمَلْ
حينما مات لذا
قالَ قد كَملْ

2 - إذ أتى من عرشِهِ
لفِدى الإنسانْ
تمَّ مسعاهُ هنا
كاملَ الإتقانْ

ق. قد قضى ديني
كلَّه الحمَلْ
حينما مات لذا
قالَ قد كَملْ

3 - أيها الساعي لأنْ
تُدرِكَ البِرّا
ثِقْ ففادي الناسِ قدْ
تمَّمَ الأمرَا

ق. قد قضى ديني
كلَّه الحمَلْ
حينما مات لذا
قالَ قد كَملْ

4 - صالحُ الأعمالِ ذا
ثمرُ الإيمانْ
لازمٌ لكنَّهُ
ما به غُفرانْ

ق. قد قضى ديني
كلَّه الحمَلْ
حينما مات لذا
قالَ قد كَملْ

5 - فإلى الفادي الْجَأوا
أيها الخطاهْ
تَلبَسوا من بِرِّهِ
حُلَّةَ الحياهْ

ق. قد قضى ديني
كلَّه الحمَلْ
حينما مات لذا
قالَ قد كَملْ
