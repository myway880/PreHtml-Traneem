1 - نشيدي يعلو بالهتافِ وكذا التمجيدْ
والسُّبحِ والتَّرنُمِ لسيدي المجيدْ
فذاك ربِّيَ الذي سُكناهُ في قلبي
فأغلِبُ وأُنشد: يسوعُ حيٌّ فـيَّ

ق. يسوعُ حيٌّ فـيَّ - يسوعُ حيٌّ فـيَّ
فذا نشيد غلبتي، يسوعُ حيٌّ فـيَّ
يسوعُ حيٌّ فـيَّ - يسوع حيٌّ فـيَّ
وذا ضمان نصرتي، يسوع حيٌّ فـيَّ

2 - إن هاجتِ البحارُ من حولي ولا نجيدْ
واشتدتِ الحربُ معَ عدُوِّيَ العنيدْ
لا أرهبُ وليس ذا بفضلِ قُوتي
بل واثقٌ لأنّهُ، يسوع حيٌّ فـيَّ

ق. يسوعُ حيٌّ فـيَّ - يسوعُ حيٌّ فـيَّ
فذا نشيد غلبتي، يسوعُ حيٌّ فـيَّ
يسوعُ حيٌّ فـيَّ - يسوع حيٌّ فـيَّ
وذا ضمان نصرتي، يسوع حيٌّ فـيَّ

3 - معَ المسيحٌ قد صلبتُ لا أنا أحيا
بل من أحَبَّني فقطْ في داخلي يحيا
شكراً لك في موكِبِ النصرِ تقودُني
يبقى شعاري دائماً : يسوع حيٌّ فـيَّ

ق. يسوعُ حيٌّ فـيَّ - يسوعُ حيٌّ فـيَّ
