1 - حبُّ السيدِ فخرُ السُّجَّدِ
حُلوٌ ثابتٌ سامِ عجيبْ
حبٌّ طاهرٌ حبٌّ وافرٌ
قوموا مجِّدوا الابنَ الحبيبْ

ق. مُلكٌ مَجدٌ قوةٌ للفادي
نادوا بالغفرانِ للأثيمْ
حبُّ الربِّ المنعشُ الفؤادِ
بابٌ للحياةِ والنعيمْ

2 - حبُّ الغافرِ ذاك الناصري
يمحو الإثمَ والشرَّ الـمَعيبْ
حبٌّ كاملٌ حبٌّ شاملٌ
بالإنشادِ كرِّموا الحبيبْ

ق. مُلكٌ مَجدٌ قوةٌ للفادي
نادوا بالغفرانِ للأثيمْ
حبُّ الربِّ المنعشُ الفؤادِ
بابٌ للحياةِ والنعيمْ

3 - حبُّ المفتدي روحُ المهتدي
للآمالِ رُكنٌ لا يخيبْ
حبٌّ حاضرٌ حبٌّ ناظرٌ
بالتهليلِ سبِّحوا الحبيبْ

ق. مُلكٌ مَجدٌ قوةٌ للفادي
نادوا بالغفرانِ للأثيمْ
حبُّ الربِّ المنعشُ الفؤادِ
بابٌ للحياةِ والنعيمْ

4 - حبُّ المنتصرْ، عونُ المنكسِرْ
يحمي النفسَ من موتٍ مُريبْ
حبٌّ نافعٌ حبٌّ شافعٌ
كلُّ الحمدِ يهدَى للحبيبْ

ق. مُلكٌ مَجدٌ قوةٌ للفادي
نادوا بالغفرانِ للأثيمْ
