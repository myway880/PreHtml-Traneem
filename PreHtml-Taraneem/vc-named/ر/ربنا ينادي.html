1 - ربُّنا يُنادي
يا كلَّ العِبادِ
ها المسيحُ الفادي
يطلبُ الخُطاهْ
إرجِعوا إليهِ
وامكُثوا لديهِ
تجنوا من يدَيهِ
جِدَّةَ الحياهْ

ق. هيا للمسيحْ
أيها البعيدْ
وتعال حالاً
تُصبِحَنْ سعيدْ

2 - يمنحُ الرجاءَ
يُرسِلُ العزاءَ
يسكنُ السماءَ
يشفعُ لنا
هيا لا تُؤجِّلْ
أسرِعنَّ أقبِلْ
في الطريقِ هلِّلْ
قد خلَّصَنا

ق. هيا للمسيحْ
أيها البعيدْ
وتعال حالاً
تُصبِحَنْ سعيدْ

3 - إذهبَنَّ خبِّرْ
بالحياةِ بشِّرْ
للجموعِ واذكُرْ
صُنعَهُ العجيبْ
كم هو كريمٌ
قلبُه رحيمٌ
حبُّه عظيمٌ
دوماً يستجيبْ

ق. هيا للمسيحْ
أيها البعيدْ
وتعال حالاً
تُصبِحَنْ سعيدْ
