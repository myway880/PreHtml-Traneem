1 - يا له نبعاً طهوراً
قد جرى للاغتسالْ
فتَسارَعْتُ سروراً
غاسلاً عني الضَّلالْ

ق. كلُّ شكري للمسيحِ
ومديحي للحَمَلْ
دمُ فاديَّ الذَّبيحِ
وصمةَ الآثامِ قد غسلْ

2 - غَفرَ الفادي ذُنوبي
ونَفي عني الشَّقاءْ
فلِفاديَّ الحبيبِ
كلُّ شكرٍ وثَناءْ

ق. كلُّ شكري للمسيحِ
ومديحي للحَمَلْ
دمُ فاديَّ الذَّبيحِ
وصمةَ الآثامِ قد غسلْ

3 - نابِذاً كلَّ الأماني
والمساعي والطُّموحْ
كلُّ ما أبغي التفاني
في رِضَى الفادي الصَّفوحْ

ق. كلُّ شكري للمسيحِ
ومديحي للحَمَلْ
دمُ فاديَّ الذَّبيحِ
وصمةَ الآثامِ قد غسلْ

4 - سبِّحوا الآبَ الرحيما
باركوا الابنَ الوحيدْ
طاوعوا الروحَ الكريما
فهو ثالوثٌ مجيدْ
