1 - هل يستطيعُ الربِّ بي
أن يصنعَ العجائبَ
وإن طلبتُ تكريسي
هل يستجيبُ الطّلبَ

ق. نعمْ نعمْ نعمْ يقولُ ربُّنا
تقدسوا للعملِ
غداً سأعملُ بكمْ
في وسطِكُمْ
في وسطِكم عجائبي

2 - هل يستطيع الربُّ أنْ
يقدِّسَ ذا الجسدا
مع ضعفِه وإثمِهِ
به يحِلُّ للمدى

ق. نعمْ نعمْ نعمْ يقولُ ربُّنا
تقدسوا للعملِ
غداً سأعملُ بكمْ
في وسطِكُمْ
في وسطِكم عجائبي

3 - هل يستطيعُ الرب أن
يُشغلَ مواهبي
مقدِّساً عواطفي
لمجد مَن أحبني

ق. نعمْ نعمْ نعمْ يقولُ ربُّنا
تقدسوا للعملِ
غداً سأعملُ بكمْ
في وسطِكُمْ
في وسطِكم عجائبي

4 - هل يستطيع الربُّ أنْ
يُبارِكَ في خِدمتي
مكرِّساً لمجدِهِ
كُلي له بجُملتي

ق. نعمْ نعمْ نعمْ يقولُ ربُّنا
تقدسوا للعملِ
