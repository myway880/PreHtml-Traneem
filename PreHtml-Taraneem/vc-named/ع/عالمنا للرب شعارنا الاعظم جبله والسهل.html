1 - عالَمُنا للربِّ
شِعارُنا الأعظمْ
جبلُه والسَّهلُ
وبرُّه والـيَمْ
عالـمُنا للربِّ
وكلُّ شيءٍ فيهْ
له سيجثو الكلُّ
قاصيهِ معْ دانيهِ

ق. هيا اعلنوا للعالمِ
بُشرى السلامْ
وباسمه لِيَعترفْ
كلُّ الأنامْ

2 - عالـمُنا للربِّ
ألمبدِعِ الأكوانْ
بدمِه العجيبِ
فدى بني الإنسانْ
عالـمُنا للربِّ
إذاً فلا نخورْ
فالنصرُّ للصليبِ
على مدى الدهورْ

ق. هيا اعلنوا للعالمِ
بُشرى السلامْ
وباسمه لِيَعترفْ
كلُّ الأنامْ

3 - عالـمُنا للربِّ
فأمرُه الجليلْ
هيّا اذهبوا ونادوا
للكلِّ بالإنجيلْ
عالـمُنا للربِّ
ذا بَندُنا الخفّاقْ
نسعى بكلِّ عزمٍ
كي نَبلُغَ الآفاقْ

ق. هيا اعلنوا للعالمِ
بُشرى السلامْ
وباسمه لِيَعترفْ
كلُّ الأنامْ
