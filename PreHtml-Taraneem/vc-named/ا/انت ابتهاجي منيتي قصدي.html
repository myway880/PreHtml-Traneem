1 - أنت ابتهاجي مُنيتي قصدي
مولايَ زِدْ حُبي وإركاني
نفسي لقد قرّت على العهدِ
ما لي سِواك فزِدْ ربي إيماني

ق. آمنتُ يا ربي فقوِّ إيماني
شدِّدْ يقيني وزِد فيك إركاني

2 - مولايَ لست للرضى أهلا
لكِنْ فِدى فاديَّ نجَّاني
رفقاً بضَعفي أيُّها المولى
آمنتُ قوِّ وزِد ربي إيماني

ق. آمنتُ يا ربي فقوِّ إيماني
شدِّدْ يقيني وزِد فيك إركاني

3 - إني لَوَانٍ فاتِرُ الحبِّ
لكنَّ لي ينبوعَ إحسانِ
في صدرِكَ الراوي صدى القلبِ
مولايَ زِد بك حُبي وإيماني

ق. آمنتُ يا ربي فقوِّ إيماني
شدِّدْ يقيني وزِد فيك إركاني

4 - زِدني اقتراباً منك في النفسِ
أنت الذي بالحبِّ ترعاني
مِن برِّك الأنوارُ كالشمسِ
ضاءَت فبارِك وزِد ربي إيماني

ق. آمنتُ يا ربي فقوِّ إيماني
شدِّدْ يقيني وزِد فيك إركاني

5 - مولاي إني مؤمنٌ زدني
عزماً يُقوِّي بالفِدى أمني
واجعل يقيني ثابتَ الرُّكنِ
واشدُد بروحك مولايَ إيماني

ق. آمنتُ يا ربي فقوِّ إيماني
شدِّدْ يقيني وزِد فيك إركاني
