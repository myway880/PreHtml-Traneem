1 - أقبِلْ إلى الفادي
يا أيها الضعيفْ
فالقُوةُ العُظمى
بعونِهِ اللَّطيفْ

ق. مُفتدي الأنامْ
قد محا الآثامْ
جاعلاً قِرمِزَها
كالثلجِ في الأعلامْ

2 - إنَّ دمَ الفادي
يُغَسِّلُ الذنوبْ
وروحَه القُدسي
يُطهِّرُ القلوبْ

ق. مُفتدي الأنامْ
قد محا الآثامْ
جاعلاً قِرمِزَها
كالثلجِ في الأعلامْ

3 - من مات عن إثمي
صليبُهُ فَخري
إذ ليس لي برٌّ
فموتُهُ بِرّي

ق. مُفتدي الأنامْ
قد محا الآثامْ
جاعلاً قِرمِزَها
كالثلجِ في الأعلامْ

4 - ألقاهُ حينما
يأتي من السماءْ
إليه أرتقي
في قُبَّةِ الهواءْ

ق. مُفتدي الأنامْ
قد محا الآثامْ
جاعلاً قِرمِزَها
كالثلجِ في الأعلامْ

5 - في مجدِهِ الأعلى
أكون كلَّ حينْ
إذ قد محا إثمي
بدمِه الثمينْ

ق. مُفتدي الأنامْ
قد محا الآثامْ
جاعلاً قِرمِزَها
كالثلجِ في الأعلامْ
