1 - إن تبِعنا الحبيبْ
حاملينَ الصليبْ
بخضوعٍ في وادي الدموعْ
إذا حُزنا رِضاهْ
نسكُنُ في حِماهْ
آمنينَ دوماً في يسوعْ

ق. ثِقْ بيسوعْ
طائعاً بالخشوعْ
فتعيشَ سعيداً
إن تثِقْ بيسوعْ

2 - إن بحُزنٍ عَميقْ
أو بهَمٍّ وضيقْ
لست عن ربي أبغي الرُّجوعْ
أَصغي الى نِداهْ
أسلُك في رِضاهْ
واثقاً دائماً بيسوعْ

ق. ثِقْ بيسوعْ
طائعاً بالخشوعْ
فتعيشَ سعيداً
إن تثِقْ بيسوعْ

3 - لا نعي حبَّه
لا نرى مجدَهُ
الا إنْ سلَّمْنا بخضوعْ
فتَجِي البركاتْ
من سما السمواتْ
إن وثِقْنا دوماً بيسوعْ

ق. ثِقْ بيسوعْ
طائعاً بالخشوعْ
فتعيشَ سعيداً
إن تثِقْ بيسوعْ

4 - حولَ ربِّ الأنامْ
والكلُّ في سلامْ
نعبُدُ فادينا بخشوعْ
بل وراهُ نسيرْ
في الطريقِ المنيرْ
واثقينَ دوماً بيسوعْ

ق. ثِقْ بيسوعْ
طائعاً بالخشوعْ
فتعيشَ سعيداً
