1. تِهتُ عنِ القطيعْ
مثلَ الخروفِ الضالْ
ينأى عن الراعي الوديعْ
للقفرِ والجبالْ
مُهاجِرَ الأوطانْ
كالولَدِ الشاردْ
بل تِهتُ في قفرِ الهوانْ
عن ذلك الوالدْ

2. راعيَّ في الحقولْ
قد نَشَدَ الخروفْ
وقد دعا ابنَهُ الجهولْ
الوالِدُ الرؤوفْ
مضى إلى الجبالْ
وفتشَ الوِديانْ
ثم رآني في خَبالْ
كالفاقدِ الوِجدانْ

3. فساقَني على
مناهِجِ الوَدادْ
فصِرتُ مُنقاداً إلى
جميعِ ما أرادْ
راعي الخروفِ الضالْ
يسوعُ ذو الإنعامْ
يرُدُّني عنِ الضلالْ
إلى حِمى السلامْ

4. نفسي عن المسيحْ
تَتيهُ في القِفارْ
تهيمُ في البرِّ الفسيحْ
في حالةِ النِّفارْ
لما أحبَّها
مُخلصي القديرْ
قد شاءَ أن يرُدَّها
