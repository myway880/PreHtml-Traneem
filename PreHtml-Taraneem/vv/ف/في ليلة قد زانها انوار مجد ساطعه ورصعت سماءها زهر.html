1. في ليلةٍ قد زانَها
أنوارُ مجدٍ ساطعهْ
ورصَّعتْ سماءَها
زهرُ النجومِ اللامعهْ
بدا ملاكِ في العُلى
مُبشّراً أهلَ الثرى
أنَّ المسيحَ قد أتى
من مجدهِ - يفدي الورى

2. فظهرت جندُ السما
تشدو بأنغامِ الحبورْ
لله مجدٌ في العُلى
وبالملا كلُّ السرورْ
في الأرضِ قد ساد السلامْ
لما أتاها ربُّهُ
واللُطفُ قم عمَّ الأنامْ
بمن تسامى حبُّهُ

3. وافى مِنَ الشرقِ المجوسْ
لرُؤيةِ الطفلِ الكريمْ
أحيَوْا بمرآهُ النفوسْ
وسبَّحوا اللهَ العظيمْ
وقدموا مِن الكُنوزْ
مُرّاً لُباناً وذهبْ
لمن بهِ كلُّ الرموزْ
تمَّتْ بفائِقِ العجَبْ

4. يا أيها الطفلُ العجيبْ
يا ابنَ الإلهِ الأزلي
أنت عزائي والطبيبْ
في كل خطبٍ وجللْ
وأنت فاديَّ الحبيبْ
وسيدي ومُرشدي
قلبي لك المثوى الرحيبْ
