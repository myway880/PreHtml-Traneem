1 - نحو قبر الرب أدنو
تاركاً خلفي الحياةْ
تاركاً همي وإثمي
ذاكراً حبَّ الإلهْ
كم نسيمُ الحب عطرٌ
ينعشُ قلبَ الخُطاهْ
قد بكي قلبي، ودمعي
من خلالِ العينِ تاهْ

2 - أنظروا فالشمسُ غابتْ
وطغى الليلُ العجيبْ
أنظروا الربَّ يسوعَ
إنه فوق الصليبْ
آهِ مِن قومٍ قساةٍ
سفكوا دمَ الحبيبْ
إن نفسي في مرارٍ
حُزنُها نارٌ تُذيبْ

3 - ألبَسوهُ تاج شوكٍ
لطموهُ جرحوهْ
وسقَوهُ الخلَّ مُرّاً
للصليب ثبَّتوهْ
كيف أن الرب أيضاً
قبِل ما فعلوهْ
صامتاً من أجل حبي
فادياً مَن أنكروهْ

4 - أيها الباكي الحزينُ
قد بكى عنك الحبيبْ
أيها المَدينُ دينُكْ
سُدِّد فوق الصليبْ
لا تُغطّيك الآلامُ
أيها العاني الكئيبْ
لا يعودُ الهمُّ يُضني
روحَك فهو المجيبْ
