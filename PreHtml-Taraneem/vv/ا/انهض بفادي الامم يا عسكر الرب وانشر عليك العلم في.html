1. إنهض بفادي الأممْ
يا عسكرَ الربِّ
وانشر عليك العلَمْ
في موقف الحربِ
يسوعُ كنزُ النِّعَمْ
يقود للغلْبِ
وكل عاصٍ هزمْ
وساد بالحبِّ

2. إتبعْ رئيسَ العُلى
يا جيش إيمانِ
واثبُت مُقيماً على
وعدٍ وإحسانِ
فلا تكن وجِلا
وافتُك بشيطانِ
هيهاتِ أن تُخذلا
مِن عونِ رحمانِ

3. قُم مُقبلاً للحملْ
يا محفل الجندِ
واحمل سلاحاً كمَلْ
واهجم على الضدِّ
حارب بدون مللْ
لا تخش من لَحدِ
مَن ليسوعَ انتقلْ
قد فاز بالمجدِ

4. حربٌ قصيرُ الزمنْ
فاغلبْ ولا تجزعْ
فالآنَ دمعُ الحَزنْ
لكن غداً يُنزعُ
إكليلُ مجدٍ حسنْ
يُعطاهُ مَن يخضعْ
يحظى بمجدٍ الوطنْ
وينتفي المدمعْ
