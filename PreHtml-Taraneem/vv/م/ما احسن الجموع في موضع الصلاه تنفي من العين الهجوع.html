1. ما أحسَنَ الجُموعْ
في موضِعِ الصلاهْ
تَنفي مِنَ العينِ الهُجوعْ
محبةُّ الإلهْ

2. ما أحسن السُّجودْ
للهِ بالوَرَعْ
فإنه ربُّ الوُجودْ
والكلَّ قد صَنعْ

3. ما أحسن الجلوسْ
في بيتِهِ المنيرْ
تحيا به كلُّ النفوسْ
لأنه القديرْ

4. يا أيها الأنامْ
قوموا ورنِّموا
وكرِّروا على الدوامْ
حمداً وعظموا

5. أُحِبُّ أن أكونْ
بينَ المرَنِّمينْ
في حضرة الآبِ الحنونْ
هناك كل حين
