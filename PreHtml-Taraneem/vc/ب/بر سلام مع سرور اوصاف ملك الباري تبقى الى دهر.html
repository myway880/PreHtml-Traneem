1. بِرٌّ سلامٌ معْ سرورْ
أوصافُ مُلْكِ الباري
تبقى إلى دهرِ الدهورْ
للمؤمنِ المُختارِ

ق. هذا سلامُ المؤمنينْ
من ربِّهم حُلوٌ ثمينْ
سلامٌ سلامٌ
سلامٌ كلَّ حينْ

2. لا تقدِرُ الدُّنيا تُنيلْ
هذا السلامَ الأسمى
كلا ولا عنا تُزيلْ
تلك العطايا العُظمى

ق. هذا سلامُ المؤمنينْ
من ربِّهم حُلوٌ ثمينْ
سلامٌ سلامٌ
سلامٌ كلَّ حينْ

3. هذا سلامٌ لي شراهْ
ربُّ الفِدى بالصلبِ
كالنهرِ يجري في صَفاهْ
يُروي ظماءَ القلبِ

ق. هذا سلامُ المؤمنينْ
من ربِّهم حُلوٌ ثمينْ
سلامٌ سلامٌ
سلامٌ كلَّ حينْ

4. الأمنُ فيه والهجوعْ
هذا السلامُ الباقي
فإنما المعطي يسوعْ
باق على الميثاقٍ

ق. هذا سلامُ المؤمنينْ
من ربِّهم حُلوٌ ثمينْ
سلامٌ سلامٌ
سلامٌ كلَّ حينْ

5. إن تَطْمُ حولي النائباتْ
كاللُّجِّ وسْطَ البحرِ
يدُمْ سلامي في ثباتْ
