1.قد غرقتُ في الشرور
في حمأة الآثام
غاب عن قلبي السرور
والأمنُ والسلام

فدعوتُ المفتدي
ربَّ الورى المنان
انتشلني منجدي
يا ذا الحنان

ق. حبّ يسوع
من حمأة الذنوب نفسي انتشَل

2. عشتُ بائسا أثيم
وحاق بي الدُّجَى
فأتى الفادي الرحيم
بالنورِ والرجا

إنَّ قلبي ملكُهُ
ما دمتُ في الوجود
فأذيعُ حمدَه
ذاك الوَدود

ق. حبّ يسوع
من حمأة الذنوب نفسي انتشَل

3. يا غريقا في الشرور
استنجدِ الإله
إنه الربُّ الغفور
يدعوكَ للنجاه

منقذًا غَرقى الآثام
من لُجَّةِ المياه
وحده فادي الأنام
رجا الخُطاه

ق. حبّ يسوع
من حمأة الذنوب نفسي انتشَل
