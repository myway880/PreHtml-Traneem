1. أنـفـض غـبار الزمــن
مــــن فـوق إنــجـيـلـك
يا وكـت تـروح عـلـيه
قـابـل هـو يـــجـيـلـــك
كـم مـرة إنـت إلتهـيت
وعـايفه بركن الـبـيـت
ومـن سـألـتـك عـلـيـه
سـكـتـت ومـا حـجـيـت

ق. لـو تـــقـرا لـو تـــقـرا
شكَد حلوة هاي الفكرة
حتعوف إبليس وتنكره
تـبوس المسيح تـشكره

2. شـكَـد حـلـوة كـلـماتــه
و حــلــوة أقـــــوالــــه
شـكَـد مأثورة آيـــاتــه
و أحلاهـم مـعـجـزاتـه
مـــزامــيـره أمـثــالــه
و رسائـله و تـكـويـنـه
أنـفـض غـبـار الزمـن
مـــــن فـوق إنـجـيـلـك

ق. لـو تـــقـرا لـو تـــقـرا
شكَد حلوة هاي الفكرة
حتعوف إبليس وتنكره
