1 - يا أربعَ الرياحِ يا
عواصفَ الخطرْ
هُبِّي على سفينتي
فلن تَرَى الضررْ

ق. لي ملجأٌ حصينْ
ومخبأٌ أمينْ
ففي صليبِ سيدي
لي ملجأٌ حصينْ

2 - يا جندَ شرٍّ هاجِموا
نفسي بلا كللْ
فبعد جهدٍ تبذلوا
تمنون بالفشلْ

ق. لي ملجأٌ حصينْ
ومخبأٌ أمينْ
ففي صليبِ سيدي
لي ملجأٌ حصينْ

3 - أيا سلاطينَ الهوى
وقوةَ الجحيمْ
ليس لنُصرتي سِوى
صلـيبِه العظيمْ

ق. لي ملجأٌ حصينْ
ومخبأٌ أمينْ
ففي صليبِ سيدي
لي ملجأٌ حصينْ

4 - لا رُؤَساءَ الأرضِ لا
موتَ ولا حياهْ
ولا ذراعَ الأقويا
تسلِبُني رضاهْ

ق. لي ملجأٌ حصينْ
ومخبأٌ أمينْ
ففي صليبِ سيدي
لي ملجأٌ حصينْ

5 - إن قام ضدي والدي
أمي وخلاني
ففي صليبِ سيدي
بلسمُ أحزاني

ق. لي ملجأٌ حصينْ
ومخبأٌ أمينْ
ففي صليبِ سيدي
لي ملجأٌ حصينْ

6 - يا مؤمنونَ التجئوا
لصخرةِ الدهورْ
تحت دِماهُ اختبئوا
فتَغلِبوا الشرورْ

ق. لي ملجأٌ حصينْ
ومخبأٌ أمينْ
ففي صليبِ سيدي
لي ملجأٌ حصينْ
