1 - لماذا أنت منحنيةْ
ويا نفسي تئنينَ
ارتجي الله فيحييكِ
بقوتهِ تغنينَ
لماذا الخوف أعياكِ
وفيه العمرَ تُفنينَ
قريبٌ منك حاميكِ
ثقي بهِ تطمئنّينَ

ق. يسوع الحيُّ يحيا
حياةَ المجدِ فيَّ
قيامتهُ مثالي
حياتي فيه ريّا

2 - لماذا الهم أضناكِ
وماذا منه تجنينَ
عليه الكلَّ ألقيه
فقد جاء ليُغنينا
لماذا الشك يغزوكِ
وأفكارا تظنّينَ
وقبرُ مسيحنا خالٍ
وفعلُه حاضرٌ فينا

ق. يسوع الحيُّ يحيا
حياةَ المجدِ فيَّ
قيامتهُ مثالي
حياتي فيه ريّا

3 - لماذا اليأس أسكتكِ
لسؤلك لست ترنينَ
وقد قام من الموتِ
وفي يده أمانينا
لماذا الموتُ يرعبكِ
وشوكتَهُ تُعانينَ
فليسَ القبرُ مثواكِ
من الأمجادِ يُدنينا

ق. يسوع الحيُّ يحيا
حياةَ المجدِ فيَّ
قيامتهُ مثالي
حياتي فيه ريّا
