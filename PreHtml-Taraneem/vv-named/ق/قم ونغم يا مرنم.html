1 - قُم ونغِّمْ يا مرنمْ
باسم فادينا الحبيبْ
برخيم اللحن عظِّمْ
ذلك الشخصَ العجيبْ
ذلك السامي السجايا
صاحبُ المجدِ الرفيعْ
خالقُ كلِّ البرايا
هللويا هللويا لطفُه مُحيي الجميعْ
أبديٌ أزليٌ
كان من قبل الجبالْ
وهو وهو قدوسٌ زكيٌ
لابس لابسٌ ثوبَ الجلالْ
أوصنا أوصنا

2 - شمسُ برٍّ ذو جمالِ
كوكبُ الصبحِ المنيرْ
وحدَه حاوي الكمالِ
ما لهُ أصلاً نظيرْ
غالبٌ فردٌ قديرُ
لقَّبوه بالأسدْ
ملجأٌ فادٍ نصيرُ
هللويا هللويا صخرُنا الحيُّ الصمدْ
أصل داود الرحيم
نسله البر الأمينْ
حَبرُ، حبرنا السامي الكريمُ
نورُ، نور كل العالمينْ
أوصنا أوصنا

3 - قُم ونغِّمْ يا مرنمْ
باسم فادينا الحبيبْ
برخيم اللحن عظِّمْ
ذلك الشخصَ العجيبْ
برُّنا المُغني الرؤوفُ
صاحبُ اللُّطفِ البديعْ
كلمة الله العطوفُ
هللويا هللويا ربُّنا الراعي الوديع
سبحوه مجدوهُ
يا جميع المؤمنين
قدسْ قدسوه كرموه
بارك باركوه حل حين
أوصنا أوصنا
