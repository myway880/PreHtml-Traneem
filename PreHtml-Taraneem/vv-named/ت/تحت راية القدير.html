1 - تحتَ رايةِ القديرْ
للأمامْ للأمامْ
كلُّنا اليومَ نسيرْ
للأمامْ
في خُطَى الراعي الأمينْ
معْ جموعِ المؤمنينْ
فلْنُجاهِدْ مُسرعينْ
للأمامْ

2 - لا نَخَفْ هولَ الجِهادْ
للأمامْ للأمامْ
لا ولا جيشَ الفَسادْ
للأمامْ
إننا جندُ المسيحْ
نحملُ النورَ الصحيحْ
لنَقودَ العالمينْ
للأمامْ

3 - في ضِيا نجمِ الفداءْ
للأمامْ للأمامْ
نحن نمشي برجاءْ
للأمامْ
فإلى النصرِ المبينْ
أيها الجندُ الأمينْ
أقدِموا مُستَبسِلينْ
للأمامْ

4 - عَلَمُ اللهِ المجيدْ
للأمامْ للأمامْ
في لَظى الحربِ الشديدْ
للأمامْ
إرفَعوا فوقَ الملا
وبِحبِّ ووَلا
فلْنَسِرْ في كلِّ حينْ
للأمامْ
