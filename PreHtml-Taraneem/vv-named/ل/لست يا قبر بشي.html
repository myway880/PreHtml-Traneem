1 - لستَ يا قبرُ بشيْ
منكَ لا نخشى وعيدا
إنما يسوعُ حيْ
فعلينا لن تسودا - هللويا

2 - في السما الحيُّ القديرْ
فالردى بابُ الحياةِ
كيف نخشى من مَسيرْ
عبرَ وادي الظُّلُماتِ - هللويا

3 - في السما الحي الغفورْ
فلنعش طوعَ يديهِ
وإلى دهر الدهورْ
نقصُرُ المجدَ عليهِ - هللويا

4 - في السما الحيُّ الشفيعْ
فلهُ نحنُ ومنهُ
ليس شيءٌ يستطيعْ
فصلَنا يا قومُ عنهُ - هللويا

5 - في السما الحيُّ العظيمْ
سائدٌ كلَّ العوالْم
عن قريبٍ سنُقيمْ
في نعيمٍ معهُ دائمْ - هللويا

6 - في السما الحيُّ الصمدْ
شمسُ برٍّ لا تغيبُ
فاقَ أنوارَ الجَلَدْ
مجدُهُ السامي العجيبُ - هللويا
