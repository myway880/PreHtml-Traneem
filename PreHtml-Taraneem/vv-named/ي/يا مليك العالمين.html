1 - يا مليكَ العالمينْ
يا رئيسَ المؤمنينْ
سُدْ بقلبي كلَّ حينْ
بهُدَى الروحِ الأمينْ
واملِكَنّي يا معينْ
يا مليك العالمينْ

2 - أنت سلطانُ الدهورْ
أنت قَهّارُ الشرورْ
قوِّني كيلا أخورْ
أنت لي حصنٌ وسورْ
من مكايِدِ الغرورْ
أنت سلطانُ الدهورْ

3 - أنت سيدي السميعْ
من ذُرى العرشِ الرفيعْ
أعطني قلباً يُطيعْ
يا مخلصَ الجميعْ
وأعِنّي يا شفيعْ
أنت سيدي السميعْ
