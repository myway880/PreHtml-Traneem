1 - يا ربِّ أقرُبُ
فأقرُبُ
إليك يا ربِّ
وأرغبُ
في الحُزن والبلا
إليك أقرُبُ
إليك أقرُبُ
فأقرُبُ

2 - إن نِمتُ في الدُّجى
وسطَ القِفارْ
وكان مَسنَدي
بعضُ الحِجارْ
فإنني إلى
فاديَّ أقربُ
فاديَّ أقربُ
فأقربُ

3 - يا ربِّ أظهِرْ لي
بابَ السما
بوعدِكِ الجَلي
أَحْيِ الرَّجا
يا خالقي أنا
إليك أقربُ
إليك أقربُ
فأقربُ

4 - إذا ارتَقَتْ نفسي
إلى العُلى
يكونُ تَرنيمي
بينَ الملا
يا ربِّ إنني
إليك أقربُ
إليك أقربُ
فأقربُ
