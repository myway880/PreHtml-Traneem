1 - إسمُ يسوعَ قد حلا
لمسمعِ المؤمنْ
يشفي جراحَ الـمُبتَلَى
والخوفَ يَستأمنْ

2 - سلوى القلوبِ الخاشعهْ
تعزيةُ الأحزانْ
قوتُ النفوسِ الجائعهْ
ومورِدُ الظَّمآن

3 - قلبي ضعيفٌ يا يسوعْ
في الحمدِ قاصرُ
والفكرُ مرتابٌ جزوعْ
والعزمُ فاترُ

4 - أُذيعُ حبَّك الصحيحْ
ما دامَ لي حياهْ
ذكرُ اسمِك السامي يُريحْ
نفسي لدى الوفاهْ
