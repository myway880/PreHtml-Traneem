1 - إسمَعوا بُشرى الظَّفرْ
بيسوعَ الخلاصْ
أُنشروها للبَشرْ
بيسوعَ الخلاصْ
بشِّروا شتّى البِلادْ
في الجبالِ والوِهادْ
علِّموا كلَّ العِبادْ
بيسوعَ الخلاصْ

2 - إهتِفوا فوقَ البحارْ
بيسوعَ الخلاصْ
في البراري والقِفارْ
بيسوعَ الخلاصْ
اشْدُ يا بحرُ النَّشيدْ
واجعلِ الجُزْرَ تُعيدْ
وكذا الأرضَ تُشيدْ
بيسوعَ الخلاصْ

3 - رنِّموا وقتَ الكِفاحْ
بيسوعَ الخلاصْ
فيه إدراكُ النَّجاحْ
بيسوعَ الخلاصْ
بشِّروا القلبَ الجريحْ
بالدَّوا الشافي الصحيحْ
واهتِفوا فوقَ الضَّريحْ
بيسوعَ الخلاصْ

4 - حمِّلوا الريحَ الخبرْ
بيسوعَ الخلاصْ
هنِّئوا كلَّ البشرْ
بيسوعَ الخلاصْ
وليُذَعْ في كلِّ دارْ
رَجْعُ هذا الانتصارْ
وليكُنْ هذا الشِّعارْ
بيسوعَ الخلاصْ
