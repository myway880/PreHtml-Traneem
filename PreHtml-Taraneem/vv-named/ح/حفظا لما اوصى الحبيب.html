1 - حفظاً لما أوصى الحبيبْ
نذكرُ موتَه العجيبْ
وهذه الذكرى تطيبْ
حتى يجيءْ

2 - إذْ نكسِرُ الخبزَ نُعِيدْ
كيف قضى الفادي المجيد
وحبُّنا له يزيدْ
حتى يجيءْ

3 - نبعُ دمائِه انفجرْ
ونال بالموتِ الظَّفرْ
ها الكأسَ تُعلنُ الخَبرْ
حتى يجيءْ

4 - نذكرُ موتَه كما
عودتُهُ مِنَ السما
مواظبينَ دائما
حتى يجيءْ

5 - حتى يُنادينا القديرْ
ويسمعَ الموتى النَّفيرْ
إذ يُنفَخُ البوقُ الأخيرْ
لمّا يجيءْ

6 - على الرَّجا هنا نُقيمْ
إيمانَ فادينا القويمْ
نرقُبُ يومَهُ العظيمْ
حتى يجيءْ
